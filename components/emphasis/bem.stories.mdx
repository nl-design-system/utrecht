import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import clsx from "clsx";
import "./bem.css";
import README from "./README.md";

export const Template = ({ textContent = "", stressed = false, strong = false }) =>
  `<span class="${clsx(
    "utrecht-emphasis",
    stressed && "utrecht-emphasis--stressed",
    strong && "utrecht-emphasis--strong"
  )}">${textContent}</span>`;

<Meta
  title="CSS Component/Emphasis"
  argTypes={{
    textContent: {
      description: "Set the text content",
      control: "text",
    },
    stressed: {
      description: "Stressed emphasis",
      control: "boolean",
    },
    strong: {
      description: "Strong emphasis: very important or urgent content",
      control: "boolean",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
    notes: {
      UX: README,
    },
  }}
/>
