import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import "./html.scss";
import README from "./README.md";

export const StressedTemplate = ({ textContent = "" }) => `<em>${textContent}</em>`;

export const StrongTemplate = ({ textContent = "" }) => `<strong>${textContent}</strong>`;

export const Template = ({ textContent = "", stressed = false, strong = false }) => {
  let formatted = textContent;
  if (stressed) {
    formatted = StressedTemplate({ textContent: formatted });
  }
  if (strong) {
    formatted = StrongTemplate({ textContent: formatted });
  }
  return `<div class="utrecht-html">${formatted}</div>`;
};

<Meta
  title="HTML Component/Emphasis"
  argTypes={{
    textContent: {
      description: "Set the text content",
      control: "text",
    },
    stressed: {
      description: "Stressed emphasis",
      control: "boolean",
    },
    strong: {
      description: "Strong emphasis: very important or urgent content",
      control: "boolean",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
    notes: {
      UX: README,
    },
  }}
/>
