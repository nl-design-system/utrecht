import { ArgsTable, Canvas, Meta, Story } from "@storybook/blocks";
import { BadgeData } from "../../../badge-data/css/story-template.jsx";
import "../index.scss";

export const BadgeList = ({ children }) => <div className="utrecht-badge-list">{children}</div>;

export const BadgeListStory = ({ items }) => (
  <BadgeList>
    {(typeof items === "string" ? items.split(/[\s.,()&;]+/g).filter(Boolean) : Array.isArray(items) ? items : []).map(
      (item, index) => (
        <BadgeData key={index}>{item}</BadgeData>
      )
    )}
  </BadgeList>
);

<Meta
  id="css-badge-list"
  title="CSS Component/Badge list"
  component={BadgeListStory}
  argTypes={{
    items: {
      description: "Items",
      control: "text",
    },
  }}
  parameters={{
    status: {
      type: "WORK IN PROGRESS",
    },
  }}
/>
