  `<div class="utrecht-alternate-lang-nav">${languages
    .map(AlternateLangLink)
    .join('<span aria-hidden="true"> | </span>')}</div>`;

<Meta
  title="Components/Alternate Language Navigation"

  }}
  parameters={{

      transformSource: (_src, { args }) => AlternateLangLink(args),
    },
    status: "WORK IN PROGRESS",
  }}
/>

      ],
    }}

    {AlternateLangNav.bind({})}
  </Story>
