import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";

<!-- Import component and component styles -->

import "./html.scss";

<!-- Import Docs -->

import README from "./README.md";
import accessibilityGuidelines from "./docs/accessibility-guidelines.md";

export const Template = ({ external = false, href = null, textContent = "" }) =>
  `<section class="utrecht-html">
  <a href="${href === null ? "#" : href}"${external ? ' rel="external noopener noreferrer"' : ""}>${textContent}</a>
</section>`;

<Meta
  title="Semantic HTML/Link"
  argTypes={{
    textContent: {
      description: "Content of the link",
      control: "text",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
    notes: {
      UX: README,
      Accessibility: accessibilityGuidelines,
    },
  }}
/>
