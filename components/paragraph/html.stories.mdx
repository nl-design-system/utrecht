export const Template = ({ textContent = "", lead = false }) =>
  `<section class="utrecht-html">
  <p${lead ? ' class="lead"' : ""}>${textContent}</p>
</section>`;

export const TemplateMulti = ({ textContent = "" }) =>
  `<section class="utrecht-html">
  <p class="lead">${textContent}</p>
  <p>${textContent}</p>
  <p>${textContent}</p>
</section>`;
