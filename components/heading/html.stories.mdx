import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";

<!-- Import component and component styles -->

import "./html.scss";

<!-- Import Docs -->

import README from "./README.md";

export const Template = ({ level = 1, textContent = "" }) =>
  `<section class="utrecht-html">
  <h${level}>${textContent}</h${level}>
</section>`;

<Meta
  title="Semantic HTML/Heading"
  argTypes={{
    level: {
      description: "Heading level",
      type: "number",
    },
    textContent: {
      description: "Set the content of the heading",
      control: "text",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
    notes: {
      UX: README,
    },
  }}
/>
