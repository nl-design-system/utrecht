import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import clsx from "clsx";

<!-- Import component and component styles -->

import "./bem.css";

<!-- Import Docs -->

import README from "./README.md";

export const Template = ({ level = 1, textContent = "", distanced = false }) =>
  `<div role="heading" aria-level="${level}" class="${clsx(`utrecht-heading-${level}`, {
    [`utrecht-heading-${level}--distanced`]: distanced,
  })}">${textContent}</div>`;

<Meta
  title="Components/Heading"
  argTypes={{
    level: {
      description: "Heading level",
      type: "number",
    },
    textContent: {
      description: "Set the content of the heading",
      control: "text",
    },
    distanced: {
      description: "Distance the element from adjacent content",
      control: "boolean",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "ALPHA",
    },
    notes: {
      UX: README,
    },
  }}
/>

# Heading components

To use the component as a web component, make sure the component is defined by including the components `element.js` file

## Heading 1

Styling via `utrecht-heading-1` class name:

<Canvas>
  <Story
    name="Heading 1"
    args={{
      distanced: true,
      level: 1,
      textContent: "The Quick Brown Fox Jumps Over The Lazy Dog",
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
