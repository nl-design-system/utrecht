import { Meta, Story, Canvas, ArgsTable } from "@storybook/addon-docs";
import clsx from "clsx";

<!-- Import component and component styles -->

import "./bem.css";

<!-- Import Docs -->

import README from "./README.md";

export const Template = ({ textContent = "", type = null }) =>
  `<span class="${clsx(
    "utrecht-form-label",
    type === "checkbox" && "utrecht-form-label--checkbox",
    type === "radio" && "utrecht-form-label--radio"
  )}">${textContent}</span>`;

<Meta
  title="CSS Component/Form label"
  argTypes={{
    textContent: {
      description: "Set the content of the label",
      control: "text",
    },
    type: {
      description: "Set the type of the form control",
      options: ["default", "checkbox", "radio"],
      control: { type: "radio" },
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: {
      type: "WORK IN PROGRESS",
    },
    notes: {
      UX: README,
    },
  }}
/>
