import { Meta, Story, Canvas } from "@storybook/addon-docs/blocks";
import clsx from "clsx";

<!-- Import Docs -->

import README from "./README.md";

export const Template = ({ items }) => `<div class="utrecht-navhtml">
   <nav class="sidenav">
    <ul class="utrecht-sidenav__list">${items
      .map(
        ({ href, title, current, focus, children }) => `<li class="utrecht-sidenav__item">
          <a class="${clsx(
            "utrecht-sidenav__link",
            current && "utrecht-sidenav__link--current",
            focus && "utrecht-sidenav__link--focus"
          )}" href="${href}">${title}</a>
          <ul>${
            children
              ? children
                  .map(
                    ({ href, title, current, focus }) =>
                      `<li><a class="${clsx(
                        "utrecht-sidenav__link",
                        current && "utrecht-sidenav__link--current",
                        focus && "utrecht-sidenav__link--focus"
                      )}" href="${href}">${title}</a></li>`
                  )
                  .join("")
              : ""
          }</ul>
        </li>
      `
      )
      .join("")}
    </ul>
  </nav>
</div>
`;

<Meta
  title="Components/Navigatie SideNav"
  argTypes={{
    items: {
      description: "Menu items",
      control: "array",
    },
  }}
  parameters={{
    docs: {
      transformSource: (_src, { args }) => Template(args),
    },
    status: "IN DEVELOPMENT",
    notes: {
      UX: README,
    },
  }}
